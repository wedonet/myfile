#
设置Hadoop

可以通过附加下面的命令到 ~/.bashrc 文件中设置 Hadoop 环境变量。

/usr/local/hadoop/hadoop-2.8.4


export HADOOP_HOME=/usr/local/hadoop/hadoop-2.8.4

export PATH = $HADOOP_HOME/bin:$PATH

#验证

hadoop version
